Aim:- To implement Knapsack Problem.

#include <stdio.h>

int main() {
    float weight[50], profit[50], ratio[50], Totalvalue, temp, capacity, amount;
    int i, j, num;

    printf("Enter number of items: ");
    scanf("%d", &num);

    // Input weights and profits for each item
    for (i = 0; i < num; i++) {
        printf("\nEnter Weight and Profit for item[%d]: ", i);
        scanf("%f %f", &weight[i], &profit[i]);
    }

    printf("\nEnter capacity of knapsack: ");
    scanf("%f", &capacity);

    // Calculate profit-to-weight ratio for each item
    for (i = 0; i < num; i++) {
        ratio[i] = profit[i] / weight[i];
    }

    // Sort items based on their profit-to-weight ratio (descending order)
    for (i = 0; i < num; i++) {
        for (j = i + 1; j < num; j++) {
            if (ratio[i] < ratio[j]) {
                temp = ratio[j];
                ratio[j] = ratio[i];
                ratio[i] = temp;

                temp = weight[j];
                weight[j] = weight[i];
                weight[i] = temp;

                temp = profit[j];
                profit[j] = profit[i];
                profit[i] = temp;
            }
        }
    }

    // Greedy approach to solve the Knapsack Problem
    printf("\nKnapsack Problem using Greedy Method: \n");
    for (i = 0; i < num; i++) {
        if (weight[i] > capacity)
            break;
        else {
            Totalvalue += profit[i];
            capacity -= weight[i];
        }
    }
    if (i < num)
        Totalvalue += (ratio[i] * capacity);

    printf("\nThe maximum value is: %f\n", Totalvalue);

    return 0;
}

2. Initialize a table K[][] to store the maximum values that can be obtained for different weights and items.
3. Iterate over each item and each possible weight from 0 to the maximum capacity W of the knapsack:
    - If the current item's weight is greater than the current weight capacity, 
        set the value of K[i][w] to the value of K[i-1][w] (same as the previous row).
    - Otherwise, set the value of K[i][w] to the maximum of:
        - The value of the current item plus the value obtained from the remaining weight capacity (K[i-1][w - wt[i]]) 
        - The value obtained from excluding the current item (K[i-1][w]).
4. The value in K[n][W] represents the maximum value that can be obtained with the given items and weight capacity.
